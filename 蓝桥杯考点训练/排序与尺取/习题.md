# [洛谷P1177](https://www.luogu.com.cn/problem/P1177)
```c++
#include <iostream>
using namespace std;

// 1e5 + 9 对于 P1177 刚好够用
const int N = 100010;
int a[N], temp[N];

// 注意这里的参数分隔符是逗号不是分号
void merge_sort(int a[], int l, int r) {
    if (l >= r) return;

    // 1. 确定分界点，防止溢出的标准写法
    int mid = l + (r - l) / 2;

    // 2. 递归
    merge_sort(a, l, mid);
    merge_sort(a, mid + 1, r);

    // 3. 合并
    int k = 0;
    int i = l;       // 左指针从 l 开始
    int j = mid + 1; // 右指针从 mid + 1 开始

    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) temp[k++] = a[i++];
        else temp[k++] = a[j++];
    }

    // 补齐剩余部分
    while (i <= mid) temp[k++] = a[i++];
    while (j <= r) temp[k++] = a[j++];

    // 4. 将排序后的 temp 拷贝回原数组 a 的 [l, r] 区间
    for (int i = l, j = 0; i <= r; i++, j++) {
        a[i] = temp[j];
    }
}

int main() {
    // 快速 IO，在大规模数据排序时能省下不少时间
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    if (!(cin >> n)) return 0;
    for (int i = 0; i < n; i++) cin >> a[i];

    merge_sort(a, 0, n - 1);

    for (int i = 0; i < n; i++) {
        cout << a[i] << (i == n - 1 ? "" : " ");
    }
    return 0;
}
```

# [洛谷P1908](https://www.luogu.com.cn/problem/P1908)
```c++
#include <iostream>

using namespace std;

typedef long long LL;

const int N = 500010;
int a[N], temp[N];
LL ans = 0; // 记录逆序对数量，必须用 long long

void merge_sort(int l, int r) {
    if (l >= r) return;

    int mid = l + (r - l) / 2;
    
    merge_sort(l, mid);
    merge_sort(mid + 1, r);

    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) {
            temp[k++] = a[i++];
        } else {
            // 核心逻辑：a[i] > a[j]，则 a[i...mid] 都大于 a[j]
            ans += (mid - i + 1); 
            temp[k++] = a[j++];
        }
    }

    // 补齐
    while (i <= mid) temp[k++] = a[i++];
    while (j <= r) temp[k++] = a[j++];

    // 拷回
    for (int i = l, j = 0; i <= r; i++, j++) a[i] = temp[j];
}

int main() {
    // 提速
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];

    merge_sort(0, n - 1);

    cout << ans << endl;

    return 0;
}
```

# [洛谷P1638](https://www.luogu.com.cn/problem/P1638)
```c++
#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 1000005;
const int MAXM = 2005;
int a[MAXN];      // 存储画展上的画
int cnt[MAXM];    // 统计当前窗口内每位画家的作品数

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> a[i];

    int l = 1, r = 0;
    int total = 0;    // 当前窗口内画家的种类数
    int min_len = 2e9;
    int ans_l = 0, ans_r = 0;

    // 滑动窗口
    while (l <= n) {
        // 右边界向右扩展，直到集齐 m 位画家
        while (r < n && total < m) {
            r++;
            if (cnt[a[r]] == 0) total++; // 遇到新画家
            cnt[a[r]]++;
        }

        // 如果集齐了，记录答案并尝试缩小左边界
        if (total == m) {
            if (r - l + 1 < min_len) {
                min_len = r - l + 1;
                ans_l = l;
                ans_r = r;
            }
        }

        // 左边界向右移，踢出一幅画
        cnt[a[l]]--;
        if (cnt[a[l]] == 0) total--; // 踢出后某位画家没了，total 减 1
        l++;
    }

    cout << ans_l << " " << ans_r << endl;

    return 0;
}
```

# [洛谷P1177](https://www.luogu.com.cn/problem/P1177)
```c++
#include <iostream>
using namespace std;

// 1e5 + 9 对于 P1177 刚好够用
const int N = 100010;
int a[N], temp[N];

// 注意这里的参数分隔符是逗号不是分号
void merge_sort(int a[], int l, int r) {
    if (l >= r) return;

    // 1. 确定分界点，防止溢出的标准写法
    int mid = l + (r - l) / 2;

    // 2. 递归
    merge_sort(a, l, mid);
    merge_sort(a, mid + 1, r);

    // 3. 合并
    int k = 0;
    int i = l;       // 左指针从 l 开始
    int j = mid + 1; // 右指针从 mid + 1 开始

    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) temp[k++] = a[i++];
        else temp[k++] = a[j++];
    }

    // 补齐剩余部分
    while (i <= mid) temp[k++] = a[i++];
    while (j <= r) temp[k++] = a[j++];

    // 4. 将排序后的 temp 拷贝回原数组 a 的 [l, r] 区间
    for (int i = l, j = 0; i <= r; i++, j++) {
        a[i] = temp[j];
    }
}

int main() {
    // 快速 IO，在大规模数据排序时能省下不少时间
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    if (!(cin >> n)) return 0;
    for (int i = 0; i < n; i++) cin >> a[i];

    merge_sort(a, 0, n - 1);

    for (int i = 0; i < n; i++) {
        cout << a[i] << (i == n - 1 ? "" : " ");
    }
    return 0;
}
```

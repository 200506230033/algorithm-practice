# 1. __[洛谷P1008](https://www.luogu.com.cn/problem/P1008)__
```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    int double_num, triple_num;
    int a1[9];
    
    // 优化：i 的上限是 329，因为 329 * 3 = 987，再大就超过三位数了
    for (int i = 123; i <= 329; i++) {
        double_num = 2 * i;
        triple_num = 3 * i;
        
        // 分离第一个数的 百位、十位、个位
        a1[0] = i / 100;
        a1[1] = (i / 10) % 10;
        a1[2] = i % 10;
        
        // 分离第二个数的 百位、十位、个位
        a1[3] = double_num / 100;
        a1[4] = (double_num / 10) % 10;
        a1[5] = double_num % 10;
        
        // 分离第三个数的 百位、十位、个位
        a1[6] = triple_num / 100;
        a1[7] = (triple_num / 10) % 10;
        a1[8] = triple_num % 10;
        
        // 将 9 个数字放入 set 去重
        set<int> s(a1, a1 + 9);
        
        // 判断条件：
        // 1. set 的大小必须是 9 (说明 9 个数字互不相同)
        // 2. set 中不能包含数字 0 (题目要求是 1~9)
        if (s.size() == 9 && s.count(0) == 0) {
            // 注意输出格式，中间要有空格
            cout << i << " " << double_num << " " << triple_num << "\n";
        }
    }
    return 0;
}
```

# 2. __[洛谷P1149](https://www.luogu.com.cn/problem/P1149)__
```c++
#include <bits/stdc++.h>
using namespace std;

// 0~9 每个数字需要的火柴棒数量
int matches[10] = {6, 2, 5, 5, 4, 5, 6, 3, 7, 6};

// 计算任意非负整数 x 所需的火柴棒总数
int get_matches(int x) {
    // 必须单独处理 0，因为下面的 while 循环进不去
    if (x == 0) {
        return matches[0];
    }
    
    int sum = 0;
    while (x > 0) {
        sum += matches[x % 10]; // 提取当前最低位，去数组里查火柴数，并累加
        x /= 10;                // 砍掉最低位
    }
    return sum;
}

int main() {
    int n;
    cin >> n;
    
    // 如果给的火柴连加号和等号（4根）都拼不出来，直接输出 0
    if (n < 4) {
        cout << 0;
        return 0;
    }
    
    int count = 0;
    
    // 枚举 A 和 B。循环上限设为 1000 已经足够大。
    // 因为剩余最多 20 根火柴，组成 1111 需要 8 根，1111+1=1112 组合是足够覆盖极限的。
    for (int i = 0; i <= 1000; i++) {
        for (int j = 0; j <= 1000; j++) {
            // C 的值是固定的
            int k = i + j; 
            
            // 如果 A、B、C 三个数需要的火柴棒加起来，刚好等于手里的火柴棒数量
            if (get_matches(i) + get_matches(j) + get_matches(k) == n - 4) {
                count++;
            }
        }
    }
    
    cout << count;
    return 0;
}
```

# 3.__[洛谷P2010](https://www.luogu.com.cn/problem/P2010)__
```c++
#include <bits/stdc++.h>
using namespace std;

// 每个月的天数，平年（把第0个位置空出来，方便直接用1-12作索引）
int days[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

// 判断是否是闰年
bool is_leap(int year) {
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
        return true;
    }
    return false;
}

// 检查拼出来的日期是否合法
bool check_valid(int date) {
    int year = date / 10000;          // 提取前四位（年）
    int month = (date / 100) % 100;   // 提取中间两位（月）
    int day = date % 100;             // 提取最后两位（日）

    // 月份必须在 1 到 12 之间
    if (month < 1 || month > 12) return false;

    // 天数不能小于 1
    if (day < 1) return false;

    // 处理 2 月的特殊情况
    if (month == 2) {
        if (is_leap(year)) {
            if (day > 29) return false;
        } else {
            if (day > 28) return false;
        }
    } 
    // 处理其他月份
    else {
        if (day > days[month]) return false;
    }

    return true;
}

int main() {
    int date1, date2;
    cin >> date1 >> date2;

    int ans = 0;
    
    // 提取起始年份和结束年份
    int year1 = date1 / 10000;
    int year2 = date2 / 10000;

    // 只需要遍历这几千个年份
    for (int y = year1; y <= year2; y++) {
        // 根据年份，生成唯一对应的回文日期
        // 比如年份 2011：
        // 2011 * 10000 = 20110000
        // 然后把 2011 倒过来拼上去
        int d1 = y % 10;            // 个位 -> 1
        int d2 = (y / 10) % 10;     // 十位 -> 1
        int d3 = (y / 100) % 10;    // 百位 -> 0
        int d4 = y / 1000;          // 千位 -> 2
        
        // 拼成完整的 8 位数：20110000 + 1102 = 20111102
        int palindrome_date = y * 10000 + d1 * 1000 + d2 * 100 + d3 * 10 + d4;

        // 判断这个回文日期是否在题目要求的范围内，并且是一个真实的日期
        if (palindrome_date >= date1 && palindrome_date <= date2 && check_valid(palindrome_date)) {
            ans++;
        }
    }

    cout << ans << endl;
    return 0;
}
```

# 1. [洛谷P8218](https://www.luogu.com.cn/problem/P8218)
```c++
#include <iostream>
#include <vector>

using namespace std;

int main() {
    // 优化输入输出效率，算法竞赛建议加上
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;

    // 前缀和数组多开一个位置，s[i] 表示前 i 个数的和
    vector<int> a(n + 1);
    vector<long long> s(n + 1, 0); 
    
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        s[i] = s[i - 1] + a[i]; // 递推计算前缀和
    }

    int m;
    cin >> m;
    while (m--) {
        int l, r;
        cin >> l >> r;
        // 区间 [l, r] 的和 = s[r] - s[l-1]
        cout << s[r] - s[l - 1] << "\n";
    }

    return 0;
}
```

# 2. [洛谷P2004](https://www.luogu.com.cn/problem/P2004)
```c++
#include <iostream>
#include <vector>
#include <climits> // 为了使用 LLONG_MIN

using namespace std;

int main() {
    // 提高 IO 速度
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, M, C;
    cin >> N >> M >> C;

    // 1. 初始化二维 vector：N+1 行，M+1 列，初始值为 0
    // 使用 1-based indexing 可以完美处理边界，不用写一堆 if
    vector<vector<long long>> s(N + 1, vector<long long>(M + 1, 0));

    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= M; ++j) {
            int num;
            cin >> num;
            // 2. 边读入边计算前缀和
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + num;
        }
    }

    long long max_val = LLONG_MIN; // 记录最大价值
    int ans_x = 1, ans_y = 1;     // 记录结果坐标

    // 3. 遍历所有可能的左上角 (i, j)
    // 对应的右下角是 (i + C - 1, j + C - 1)
    for (int i = 1; i <= N - C + 1; ++i) {
        for (int j = 1; j <= M - C + 1; ++j) {
            int x2 = i + C - 1;
            int y2 = j + C - 1;
            
            // 使用二维前缀和公式计算当前 C*C 正方形的和
            long long current_sum = s[x2][y2] - s[i - 1][y2] - s[x2][j - 1] + s[i - 1][j - 1];

            if (current_sum > max_val) {
                max_val = current_sum;
                ans_x = i;
                ans_y = j;
            }
        }
    }

    cout << ans_x << " " << ans_y << endl;

    return 0;
}
```

# 3. [洛谷P2367](https://www.luogu.com.cn/problem/P2367)
```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 关关联，提速（竞赛必备）
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, p;
    cin >> n >> p;

    // a[i] 存原始成绩，d[i] 存差分
    // 多开 2 个空间防止 R+1 越界
    vector<int> a(n + 2, 0);
    vector<int> d(n + 2, 0);

    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        // 1. 构建差分数组：D[i] = a[i] - a[i-1]
        d[i] = a[i] - a[i - 1];
    }

    // 2. 处理 P 次区间修改
    while (p--) {
        int x, y, z;
        cin >> x >> y >> z;
        d[x] += z;       // 左端点加
        d[y + 1] -= z;   // 右端点的下一个减回去
    }

    // 3. 还原数组（差分数组的前缀和即为原数组）
    // 题目要求找最低分，我们可以在还原时直接记录最小值
    int min_grade = 1e9; // 初始设为一个很大的数
    int current_grade = 0;
    
    for (int i = 1; i <= n; ++i) {
        current_grade += d[i]; // 这里的累加过程就是还原
        if (current_grade < min_grade) {
            min_grade = current_grade;
        }
    }

    cout << min_grade << endl;

    return 0;
}
```

# 4. [洛谷P3397](https://www.luogu.com.cn/problem/P3397)
```c++
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;

    // 差分数组，多开两个空间防止 x2+1 或 y2+1 越界
    vector<vector<int>> d(n + 2, vector<int>(n + 2, 0));

    while (m--) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        
        // 二维差分操作 O(1)
        d[x1][y1]++;
        d[x1][y2 + 1]--;
        d[x2 + 1][y1]--;
        d[x2 + 1][y2 + 1]++;
    }

    // 还原矩阵：对差分数组求二维前缀和
    // 公式：s[i][j] = s[i-1][j] + s[i][j-1] - s[i-1][j-1] + d[i][j]
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            d[i][j] = d[i - 1][j] + d[i][j - 1] - d[i - 1][j - 1] + d[i][j];
            cout << d[i][j] << (j == n ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
```
